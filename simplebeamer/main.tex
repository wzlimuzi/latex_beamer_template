\begin{frame}{Algorithm}
    \begin{algorithm}[H]
        \caption{Euclid's algorithm}
        \KwData{Two nonnegative integers $a$ and $b$}
        \KwResult{Their greatest common divisor $d = \gcd(a, b)$}
        \While{$b \neq 0$}{
            $r \leftarrow a \bmod b$\;
            $a \leftarrow b$\;
            $b \leftarrow r$\;
        }
        $d \leftarrow a$\;
    \end{algorithm}

\end{frame}
